//Defining a function to check the length of the line segment formed any two points .

double line(double a, double b, double c, double d){
    double z=sqrt(pow(b-a,2)+pow(d-c,2));
    return z;
}

//Defining a function to check the perpendicular distance of any line segment from the query point.

double dist(double x, double y, double a, double b, double p, double q){
    double z= (fabs(((q-y)*(a-x))-((p-x)*(b-y))))/(sqrt(pow((a-x),2)+pow((b-y),2)));
    return z;
}

int main(){
    int m,n,i,j,k,l;
    double x[1000], y[1000], ls[1000], ds[1000], p[1000], q[1000];

//Number of elements in the list of 2-D coordinates.

    printf("Enter the number of points:\n");
    scanf("%d",&n);

    printf("Enter the points(x[i],y[i]):\n");
    for(i=0;i<n;i++){
        scanf("%lf%lf",&x[i],&y[i]);
    }

//Longest and shortest line segment.

    double lar, x11, x12, y11, y12;
    for(i=0;i<n;i++){
        for(j=i+1;j<n;j++){
            k=i+j;
            ls[k] = line(x[i],x[j],y[i],y[j]);
            if(ls[k]>lar){
                lar=ls[k];
                x11=x[i];
                x12=x[j];
                y11=y[i];
                y12=y[j];
            }
        }
    }

    double shor=lar, x21, x22, y21, y22;
    for(i=0;i<n;i++){
        for(j=i+1;j<n;j++){
            k=i+j;
            ls[k] = line(x[i],x[j],y[i],y[j]);
            if(ls[k]<shor){
                shor=ls[k];
                x21=x[i];
                x22=x[j];
                y21=y[i];
                y22=y[j];
            }
        }
    }

    printf("Enter the number of query points:\n");
    scanf("%d",&m);

    printf("Enter the query points(p[i],q[i]):\n");
    for(i=0;i<m;i++){
        scanf("%lf%lf",&p[i],&q[i]);
    }

//Ouptput 1
    
    printf("The longest line segment is ((%lf,%lf),(%lf,%lf)) with a length of %lf.\n",x11,y11,x12,y12,lar);
    printf("The shortest line segment is ((%lf,%lf),(%lf,%lf)) with a length of %lf.\n",x21,y21,x22,y22,shor);

//Checking whether query point is a member of the list of 2-D coordinate points.

    int check[m];

    for(l=0;l<m;l++){
        for(i=0;i<n;i++){
            if(p[l]==x[i] && q[l]==y[i]){
                check[l]=0;
                break;
            }
            else check[l]=1;   
        }
    }
    
    for(l=0;l<m;l++){
        if((p[l]==999)&&(q[l]==999)){
            printf("Exiting..\n");
            break;
        }
        else if(check[l]==0){
            printf("The point is a member of the list.");
            break;
        }
        else{    
            double long_p;
            for(i=0;i<n;i++){
                for(j=i+1;j<n;j++){
                    k=i+j;
                    ds[k] = dist(x[i],y[i],x[j],y[j],p[l],q[l]);
                    if(ds[k]>long_p) long_p=ds[k];
                }
            }

            double x1, x2, y1, y2, shor_p=long_p;
            k=l*100;
            for(i=0;i<n;i++){
                for(j=i+1;j<n;j++){
                ds[k] = dist(x[i],y[i],x[j],y[j],p[l],q[l]);
                if(ds[k]<shor_p){ 
                shor_p=ds[k];
                x1=x[i];
                x2=x[j];
                y1=y[i];
                y2=y[j];
              }
                k++;
                }
            }
            printf("The closest line segment is ((%lf,%lf),(%lf,%lf)) for query point (%lf,%lf) with a distance of %lf.\n",x1,y1,x2,y2,p[l],q[l],shor_p);
        }
    }

    return 0;
}

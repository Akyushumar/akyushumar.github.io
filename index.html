#include <stdio.h>

#include <math.h>

void split_real_number(double num, int* w_p, double* f_p) { 

// Getting the absolute value of the input number 

double abs_num = fabs(num); 

// Getting the whole number part by truncating the absolute value 

*w_p = (int) abs_num; 

// Getting the fractional part by subtracting the whole number part from the absolute value 

*f_p = abs_num - *w_p; 

} 

int main() {

 int N; 

printf("Enter the number of real numbers to read: "); 

scanf("%d", &N); 

double num, f_p; 

int w_p, w_sum = 0; 

double f_sum = 0.0; 

for (int i = 0; i < N; i++) { 

printf("Enter a real number: "); 

scanf("%lf", &num); 

split_real_number(num, &w_p, &f_p); 

w_sum += w_p; 

f_sum += f_p; 

} 

printf("Sum of whole number parts: %d\n", w_sum); 

printf("Sum of fractional parts: %lf\n", f_sum); 

return 0; 

}

// Function to encrypt the given string using the key
void encrypt(char* s1, char* s2, int K) {
    int len1 = strlen(s1);
    int len2 = strlen(s2);
    int k_digits = floor(log10(abs(K))) + 1; // Calculate the number of digits in K
    for (int i = 0, j = 0; i < len1; i++) {
        char c = s1[i];
        int shift = (K / (int)pow(10, j % k_digits)) % 10; // Extract the digit from K
        if (isalpha(c)) {
            if (isupper(c)) {
                s2[i] = 'A' + ((c - 'A' + shift) % 26);
            } else {
                s2[i] = 'a' + ((c - 'a' + shift) % 26);
            }
        } else {
            s2[i] = c;
        }
        j++;
    }
    s2[len1] = '\0';
}

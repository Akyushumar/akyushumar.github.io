#include <stdio.h>

long int Choose_iter(int n, int i) {
    /* Computes n choose i using iteration */
    if (i > n) {
        return 0;
    }
    if (i == 0) {
        return 1;
    }
    long int result = 1;
    for (int j = 1; j <= i; j++) {
        result *= (n - j + 1);
        result /= j;
    }
    return result;
}

long int Choose_recur(int n, int i) {
    /* Computes n choose i using recursion */
    if (i > n) {
        return 0;
    }
    if (i == 0) {
        return 1;
    }
    return (n * Choose_recur(n-1, i-1)) / i;
}
int main()
 {
    int n, k;
    printf("Enter n and k(n>=k): ");
    scanf("%d %d", &n, &k);
    int m_even = k - (k % 2);  // closest even integer of k <= k
    int m_odd = k - ((k % 2) + 1);  // closest odd integer of k <= k
    long int SE = 0;    long int SO = 0;
    for (int i = 0; i <= m_even; i += 2) {
        SE += Choose_iter(n, i);
    }
    for (int i = 1; i <= m_odd; i += 2) {
        SO += Choose_iter(n, i);
    }
    printf("n = %d, k = %d\n", n, k);
    printf("SE = %ld\n", SE);
    printf("SO = %ld\n", SO);
    return 0;
}

#include <stdio.h>
#include <string.h>
int CountVowels(char a, char vowels[]){
  for(int i = 0; i < 10; i++){
    if(a == vowels[i]) return 1;
  }
  return 0;
}
int CountConsonants(char a, char vowels[]){
  for(int i = 0; i < 10; i++){
    if(a == vowels[i]) return 0;
    if(a == ' ') return 0;
  }
  return 1;
}
int main(){
  char f[25], l[25], name[50];
  char vowels[10]={'a','e','i','o','u','A','E','I','O','U'};
  int vowel = 0, consonant = 0;
 
// Taking first name as input.
  printf("Enter first name (Use ENTER to terminate):\n");
  scanf("%s",f);
 
// Taking last name as input.
  printf("Enter last name (Use ENTER to terminate):\n");
  scanf("%s",l);
 
// Appending the two strings.
  strcpy(name,f);
  int len = strlen(name);
  name[len]=' ';
  strcat(name,l);
  printf("%s\n",name);
 
// Checking for vowels and consonants.
  len = strlen(name);
  for(int i = 0; i < len; i++){
    if(CountVowels(name[i],vowels)) vowel++;
    if(CountConsonants(name[i],vowels)) consonant++;
  }
  printf("The number of vowels and consonants respectively are %d and %d.\n",vowel,consonant);
  return 0;
}

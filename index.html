#include <stdio.h>
#include <stdlib.h>
#include <math.h>
 
//Function for dynamically allocating a 2D array of real numbers of size RxC.
double **Allocate_2D_Array(int R, int C) {
double **arr = (double **)malloc(R * sizeof(double *));
for (int i = 0; i < R; i++) {
arr[i] = (double *)malloc(C * sizeof(double));
}
return arr;
}
 
//Function for reading a 2D array of real numbers of size RxC.
void Read_2D_Array(double** A, int R, int C){
for(int i=0; i<R; i++){
for(int j=0; j<C; j++){
scanf("%lf", &A[i][j]);
}
}
printf("\n");
}
 
//Function for printing a 2D array of real numbers of size RxC.
void Print_2D_Array(double** A, int R, int C){
for(int i=0; i<R; i++){
for(int j=0; j<C; j++){
printf("%lf ", A[i][j]);
}
printf("\n");
}
printf("\n");
}
 
//Function for freeing memory space of a dynamically allocated 2D array.
void Free_2D_Array(double** A, int R){
for(int i=0; i<R; i++){
free(A[i]);
}
free(A);
}
 
//Function for multiplying two matrices.
double **Multiply_matrices(double **A1, int R1, int C1, double **A2, int R2, int C2) {
if (C1 != R2) {
return NULL; //Matrices cannot be multiplied.
}
double **result = Allocate_2D_Array(R1, C2);
for (int i = 0; i < R1; i++) {
for (int j = 0; j < C2; j++) {
result[i][j] = 0;
for (int k = 0; k < C1; k++) {
result[i][j] += (A1[i][k] * A2[k][j]);
}
}
}
return result;
}
 
//Function for computing the trace of matrix.
double Compute_trace(double** A, int R, int C){
double trace=0;
for(int i=0; i<R; i++){
for(int j=0; j<C; j++){
if(i==j) trace += A[i][j];
}
}
return trace;
}
